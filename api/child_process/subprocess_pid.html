Added in: v0.1.90
  • <integer>

Returns the process identifier (PID) of the child process.

const { spawn } = require('child_process');
const grep = spawn('grep', ['ssh']);

console.log(`Spawned child pid: ${grep.pid}`);
grep.stdin.end();
